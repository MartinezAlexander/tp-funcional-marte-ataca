data Persona = UnaPersona {
    nombre:: String,
    hijos::[ Persona]
}  | Nadie  | Alguien {dni::Int}

cuantosHijos (UnaPersona nombre hijos)  = length cuantosHijos
cuantosHijos (Nadie)  = 0 
cuantosHijos (Alguien nro) = nro `div` 10000000

cuantaDescencia persona = cuantaDescendencia (head (hijos persona)) + cuantosHijos (hijos persona !!2)
